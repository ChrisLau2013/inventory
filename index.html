function manualSave(){ if(!locationSelect.value) return showToast('Select location','warn'); const prod=mProd.value.trim(); const unit=mUnit.value.trim(); const sup=mSup.value.trim(); const qty=parseInt(mQty.value); if(!prod||!unit||isNaN(qty)) return showToast('Complete fields','warn'); addProduct('Temporary',prod,unit,sup); const c=data.catalogs.find(x=>x.name==='Temporary'); const p=c.products.find(x=>x.name===prod&&x.supplier===sup); p.packages[0].records.push({ qty, location: locationSelect.value, time: new Date().toLocaleString() }); save(); refresh(); showToast('Temporary saved'); // ✅ 只清手动输入内容 mProd.value=''; mUnit.value=''; mSup.value=''; mQty.value=''; }
